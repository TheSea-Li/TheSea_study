# 一、两数之和
### 题目
![](Images/question1.png)
### 题解
```js
var twoSum = function (nums, target) {
    for (let i = 0; i < nums.length; i++) {
    for (let j = i + 1; j < nums.length; j++) {
      if (nums[i] + nums[j] === target) {
        return [i, j];
      }
    }
  }
  return [];
};
```

# 二、罗马数字转整数（字符串，数字，哈希表）
### 题目
![](Images/question2.png)
### 题解
```js
var romanToInt = function (s) {
    const map = {
        'I': 1,
        'V': 5,
        'X': 10,
        'L': 50,
        'C': 100,
        'D': 500,
        'M': 1000
    };

    let total = 0;

    for (let i = 0; i < s.length; i++) {
        const current = map[s[i]];
        const next = map[s[i + 1]];
        if (current < next) {
            total -= current;
        } else {
            total += current;
        }
    }

    return total;
};
```

# 三、 移动零
### 题目
![](Images/question3.png)
### 题解
```js
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var moveZeroes = function (nums) {
    if (!nums.includes(0)) return nums;
    let a = 0;
    for (let i = 0; i < nums.length; i++) {
        if (nums[i] != 0) {
            nums[a] = nums[i];
            a++;
        }
    }
    for (let j = a; j < nums.length; j++) {
        nums[j] = 0;
    }
    return nums;
};
```

# 四、 回文数
### 题目
![](Images/question4.png)
### 题解
```js
var isPalindrome = function (x) {
    if (x < 0) return false
    if (x == 0) return true
    let y = String(x)
    for (let i = 0; i < y.length; i++) {
        if (y[i] != y[y.length - 1 - i]) return false
    }
    return true
};
```

# 五、 最短公共前缀
### 题目
![](Images/question5.png)
### 题解
```js
var longestCommonPrefix = function(strs) {
    if (strs.length === 0) return "";
    for (let i = 0; i < strs[0].length; i++) {
        const ch = strs[0][i];
        for (let j = 1; j < strs.length; j++) {
            // 检查其他字符串长度是否足够，或字符不匹配
            if (i >= strs[j].length || strs[j][i] !== ch) {
                return strs[0].slice(0, i);
            }
        }
    }
    return strs[0];
};
```
