1. 数组合并(concat)
arr1 = [1,2,3],arr2=[2,3,4]
arr3 = arr1.concat(arr2) = [1,2,3,2,3,4]

2. 数组转字符串(join)
```js
const arr = ['apple', 'banana', 'orange'];

// 默认用逗号分隔
arr.join();        // "apple,banana,orange"

// 指定分隔符
arr.join('');      // "applebananaorange"
arr.join(' ');     // "apple banana orange"
arr.join('-');     // "apple-banana-orange"
arr.join(' and '); // "apple and banana and orange"
```

3. 字符串转数组(split)
```js
const str = 'apple,banana,orange';

// 按分隔符分割
str.split(',');        // ['apple', 'banana', 'orange']
str.split('');         // ['a','p','p','l','e',',','b','a',...]
str.split(' ');        // ['apple,banana,orange']（无空格时不分割）

// 其他分隔符
'1-2-3'.split('-');    // ['1', '2', '3']
'hello world'.split(' '); // ['hello', 'world']
```
